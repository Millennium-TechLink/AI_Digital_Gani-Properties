# ============================================
# Netlify Deployment Environment Variables
# ============================================
# Copy this file to .env and fill in your values
# DO NOT commit .env to git (it's already in .gitignore)

# ============================================
# Frontend Variables (VITE_ prefix)
# ============================================
# These are used by the React/Vite frontend

# Site URL - Your Netlify site URL
VITE_SITE_URL=https://your-site.netlify.app

# Supabase Configuration (Frontend)
VITE_SUPABASE_URL=https://your-project.supabase.co
VITE_SUPABASE_ANON_KEY=your-supabase-anon-key

# Google Sheets Contact Form (Optional)
VITE_GOOGLE_SHEETS_WEB_APP_URL=https://script.google.com/macros/s/YOUR_SCRIPT_ID/exec

# ============================================
# Backend Variables (No VITE_ prefix)
# ============================================
# These are used by Netlify Functions (server-side only)

# Supabase Configuration (Backend)
SUPABASE_URL=https://your-project.supabase.co
SUPABASE_ANON_KEY=your-supabase-anon-key
SUPABASE_SERVICE_KEY=your-supabase-service-key

# JWT Authentication
JWT_SECRET=your-secret-jwt-key-change-this-in-production
JWT_EXPIRES_IN=24h

# Admin Credentials
ADMIN_USERNAME=admin
ADMIN_PASSWORD=your-secure-admin-password

# ============================================
# Dashboard Variables (for pipeline/dashboard)
# ============================================
# These are for the separate dashboard deployment

# Dashboard API URL (points to main site's functions)
VITE_API_URL=https://your-main-site.netlify.app/.netlify/functions

# Dashboard Supabase (same as main site)
VITE_SUPABASE_SERVICE_KEY=your-supabase-service-key

# ============================================
# Optional: Form Service Configuration
# ============================================
GP_FORM_ENDPOINT=https://your-form-endpoint.com/submit
GP_FORM_METHOD=POST
GP_FORM_SECRET=your-access-key-or-token

# ============================================
# Optional: CAPTCHA (for spam protection)
# ============================================
GP_HCAPTCHA_SITEKEY=your-hcaptcha-site-key
GP_HCAPTCHA_SECRET=your-hcaptcha-secret
GP_TURNSTILE_SITEKEY=your-turnstile-site-key
GP_TURNSTILE_SECRET=your-turnstile-secret

# ============================================
# Notes:
# ============================================
# 1. Frontend variables MUST have VITE_ prefix
# 2. Backend variables (for Netlify Functions) should NOT have VITE_ prefix
# 3. After creating .env, you can use the script to set them in Netlify:
#    npm run netlify:env:set
# 4. Or manually set them in Netlify dashboard:
#    Site settings > Environment variables
